88static void escribirNodo(FILE *f, Nodo *nodo) {
89 fwrite(nodo->nombre, sizeof(char), MAX_NOMBRE, f);
90 int tipo = (int)nodo->tipo;
91 fwrite(&tipo, sizeof(int), 1, f);
92 fwrite(&nodo->numHijos, sizeof(int), 1, f);
93
94 if (nodo->tipo == ARCHIVO) {
95 fwrite(&nodo->size, sizeof(int), 1, f);
96 if (nodo->size > 0 && nodo->contenido)
97 fwrite(nodo->contenido, sizeof(char), nodo->size + 1, f);
98 else {
99 char vacio = '\0';
100 fwrite(&vacio, sizeof(char), 1, f);
101 }
102 }
103
104 for (int i = 0; i < nodo->numHijos; i++)
105 escribirNodo(f, nodo->hijos[i]);
106}

111static Nodo *leerNodo(FILE *f) {
112 char nombre[MAX_NOMBRE];
113 if (fread(nombre, sizeof(char), MAX_NOMBRE, f) != MAX_NOMBRE)

122 Nodo *nodo = crearNodo(nombre, (TipoNodo)tipo);
123
124 if (tipo == (int)ARCHIVO) {
125 int size = 0;
126 fread(&size, sizeof(int), 1, f);
127 nodo->size = size;
128 nodo->contenido = (char *)malloc(size + 2);
129 if (nodo->contenido)
130 fread(nodo->contenido, sizeof(char), size + 1, f);
131 }
132
133 for (int i = 0; i < numHijos; i++) {
134 Nodo *hijo = leerNodo(f);
135 if (hijo) agregarHijo(nodo, hijo);
136 }
137 return nodo;
