18void construirRutaAbsoluta(Nodo *nodo, Nodo *root, char *buf, int bufSize) {
19 if (nodo == root) { strncpy(buf, "/", bufSize); return; }
20
21 char piezas[32][MAX_NOMBRE];
22 int count = 0;
23 Nodo *cur = nodo;
24 while (cur != NULL && cur != root) {
25 strncpy(piezas[count++], cur->nombre, MAX_NOMBRE - 1);
26 cur = cur->padre;
27 }
28
29 buf[0] = '\0';
30 for (int i = count - 1; i >= 0; i--) {
31 strncat(buf, "/", bufSize - strlen(buf) - 1);
32 strncat(buf, piezas[i], bufSize - strlen(buf) - 1);
33 }
34 if (strlen(buf) == 0) strncpy(buf, "/", bufSize);
35}

42Nodo *resolverRuta(const char *ruta, Nodo *actual, Nodo *root) {
43 Nodo *cursor = (ruta[0] == '/') ? root : actual;
44
45 char copia[512];
46 strncpy(copia, ruta, sizeof(copia) - 1);
47 copia[sizeof(copia) - 1] = '\0';
48
49 char *token = strtok(copia, "/");
50 while (token != NULL) {
51 if (strcmp(token, ".") == 0) {
52 // Permanece en el mismo nodo
53 } else if (strcmp(token, "..") == 0) {
54 if (cursor->padre != NULL) cursor = cursor->padre;
55 } else {
56 cursor = buscarHijo(cursor, token);
57 if (cursor == NULL) return NULL;
58 }
59 token = strtok(NULL, "/");
60 }
61 return cursor;
62}
